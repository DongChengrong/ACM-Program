#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
int main()
{
    int n;
    cin >> n;
    vector<long long> ans;
    while (n--)
    {
        int a;
        cin >> a;
        ans.push_back(a);
    }
    sort(ans.begin(), ans.end());
    int j = ans[0];
    int i = 1;
    int cnt =0;
    while (i < ans.size())
    {
        cnt+=ans[i]-ans[i-1]-1;
        i++;
    }
    cout << cnt;
    return 0;
}
